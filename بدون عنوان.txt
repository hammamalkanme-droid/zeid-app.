        <h3>📊 حاسبة النسبة المئوية</h3>
        <div class="input-group">
            <label>المجموع الكلي للمواد:</label>
            <input type="number" id="totalMarks" value="280" placeholder="مثلاً: 280">
        </div>
        <div class="input-group">
            <label>درجاتك الحالية / المتوقعة:</label>
            <input type="number" id="myMarks" placeholder="أدخل درجاتك">
        </div>
        <button onclick="calculatePercentage()">احسب النسبة المئوية</button>
        <div id="resultBox" class="result success"></div>
    </div>

    <script>
        // تهيئة الـ WebApp وتطبيق ثيم التليجرام
        const tg = window.Telegram.WebApp;
        if (tg) {
            tg.ready();
            document.body.classList.add('tg');
        }

        // 1. منطق العداد التنازلي (تاريخ الامتحانات الافتراضي 15 يوليو 2026)
        const targetDate = new Date("July 15, 2026 09:00:00").getTime();

        const timer = setInterval(function() {
            const now = new Date().getTime();
            const diff = targetDate - now;

            if (diff <= 0) {
                clearInterval(timer);
                document.getElementById("countdown").innerHTML = "<b>بالتوفيق والنجاح لجميع الطلبة! 🎉</b>";
                return;
            }

            const days = Math.floor(diff / (1000 * 60 * 60 * 24));
            const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((diff % (1000 * 60)) / 1000);

            document.getElementById("days").innerText = days < 10 ? "0" + days : days;
            document.getElementById("hours").innerText = hours < 10 ? "0" + hours : hours;
            document.getElementById("minutes").innerText = minutes < 10 ? "0" + minutes : minutes;
            document.getElementById("seconds").innerText = seconds < 10 ? "0" + seconds : seconds;
        }, 1000);

        // 2. منطق حساب النسبة المئوية
        function calculatePercentage() {
            const total = parseFloat(document.getElementById("totalMarks").value);
            const mine = parseFloat(document.getElementById("myMarks").value);
            const resultBox = document.getElementById("resultBox");

            if (!total || !mine || mine > total || mine < 0) {
                alert("الرجاء إدخال درجات صحيحة ومطابقة للمجموع!");
                return;
            }

            const percentage = ((mine / total) * 100).toFixed(2);
            let estimation = "";

            if (percentage >= 85) estimation = "ممتاز 🔥";
            else if (percentage >= 75) estimation = "جيد جداً ✨";
            else if (percentage >= 65) estimation = "جيد 👍";
            else if (percentage >= 50) estimation = "مقبول 🙏";
            else estimation = "ضعيف، شد حيلك أكثر 💪";

            resultBox.style.display = "block";
            resultBox.innerHTML = `نسبتك هي: ${percentage}% <br> التقدير: ${estimation}`;
        }
    </script>
